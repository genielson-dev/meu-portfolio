const projetos = [
  {
    nome: "Projeto 1",
    descricao: "Meu primeiro projeto"
  },
  {
    nome: "Projeto 2",
    descricao: "Outro projeto simples"
  }
];

const container = document.getElementById("lista-projetos");

projetos.forEach(p => {
  const div = document.createElement("div");
  div.classList.add("card");

  div.innerHTML = `
    <h3>${p.nome}</h3>
    <p>${p.descricao}</p>
  `;

  container.appendChild(div);
});
